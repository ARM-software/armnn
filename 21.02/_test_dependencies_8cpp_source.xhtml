122 // Tests that a simple cycle in the tensorflow graph will be detected and an exception thrown, rather than the TfParser
123 // getting stuck in an infinite loop.

134  type: DT_FLOAT

136  }
137 }

145  type: DT_FLOAT
146  }
147  }
148 }
149  )";
