59 std::vector<TensorShape> BatchToSpaceNdLayer::InferOutputShapes(const std::vector<TensorShape>& inputShapes) const
60 {
61  ARMNN_ASSERT(inputShapes.size() == 1);
62 
63  const TensorShape& inputShape = inputShapes[0];
64  TensorShape outputShape(inputShape);
65 
66  unsigned int accumulatedBlockShape = std::accumulate(m_Param.m_BlockShape.begin(),
67  m_Param.m_BlockShape.end(),
68  1U,
69  std::multiplies<>());
70 
71  ARMNN_ASSERT(inputShape[0] % accumulatedBlockShape == 0);
72 
73  outputShape[0] = inputShape[0] / accumulatedBlockShape;
74 
75  DataLayoutIndexed dimensionIndices = m_Param.m_DataLayout;
76  unsigned int heightIndex = dimensionIndices.GetHeightIndex();
77  unsigned int widthIndex = dimensionIndices.GetWidthIndex();
78 
79  unsigned int heightCrop = m_Param.m_Crops[0].first + m_Param.m_Crops[0].second;
80  unsigned int widthCrop = m_Param.m_Crops[1].first + m_Param.m_Crops[1].second;
81 
82  unsigned int outputHeight = inputShape[heightIndex] * m_Param.m_BlockShape[0];
83  unsigned int outputWidth = inputShape[widthIndex] * m_Param.m_BlockShape[1];
84 
85  ARMNN_ASSERT_MSG(heightCrop <= outputHeight,
86  "BatchToSpaceLayer: Overall height crop should be less than or equal to the uncropped output height.");
87 
88  ARMNN_ASSERT_MSG(widthCrop <= outputWidth,
89  "BatchToSpaceLayer: Overall width crop should be less than or equal to the uncropped output width.");
90 
91  outputShape[heightIndex] = outputHeight - heightCrop;
92  outputShape[widthIndex] = outputWidth - widthCrop;
93 
94  return std::vector<TensorShape>({ outputShape });
95 }
