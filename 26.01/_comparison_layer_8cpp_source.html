36std::vector<TensorShape> ComparisonLayer::InferOutputShapes(const std::vector<TensorShape>& inputShapes) const
37{
38 if (inputShapes.size() != 2)
39 {
40 throw armnn::Exception("inputShapes' size is \"" + std::to_string(inputShapes.size()) +
41 "\" - should be \"2\".");
42 }
43
44 TensorShape input0 = inputShapes[0];
45 TensorShape input1 = inputShapes[1];
46
47 if (inputShapes[0].GetNumDimensions() < inputShapes[1].GetNumDimensions())
48 {
49 input1 = inputShapes[0];
50 input0 = inputShapes[1];
51 }
52 unsigned int numDims = input0.GetNumDimensions();
53 unsigned int shiftedDims = input0.GetNumDimensions() - input1.GetNumDimensions();
54
55 // Get the max of the inputs.
56 std::vector<unsigned int> dims(numDims);
57 for (unsigned int i = shiftedDims; i < numDims; i++)
58 {
59 unsigned int dim0 = input0[i];
60 unsigned int dim1 = input1[i - shiftedDims];
61
62 // Validate inputs are broadcast compatible.
63 if (dim0 != dim1 && dim0 != 1 && dim1 != 1)
64 {
65 throw armnn::Exception("Dimensions should either match or one should be of size 1.");
66 }
67
68 dims[i] = std::max(dim0, dim1);
69 }
70
71 // Fill in the rest of the shifted dimensions.
72 for (unsigned int i = 0; i < shiftedDims; i++)
73 {
74 dims[i] = input0[i];
75 }
76
77 return std::vector<TensorShape>({ TensorShape(numDims, dims.data()) });
78}

80void ComparisonLayer::ValidateTensorShapesFromInputs()
81{
82 VerifyLayerConnections(2, CHECK_LOCATION());
83
84 const TensorShape& outputShape = GetOutputSlot(0).GetTensorInfo().GetShape();
85
86 VerifyShapeInferenceType(outputShape, m_ShapeInferenceMethod);
87
88 std::vector<TensorShape> inferredShapes = InferOutputShapes({
89 GetInputSlot(0).GetTensorInfo().GetShape(),
90 GetInputSlot(1).GetTensorInfo().GetShape()
91 });
92
93 if (inferredShapes.size() != 1)
94 {
95 throw armnn::LayerValidationException("inferredShapes has "
96 + std::to_string(inferredShapes.size()) +
97 " elements - should only have 1.");
98 }
99
100 ValidateAndCopyShape(outputShape, inferredShapes[0], m_ShapeInferenceMethod, "ComparisonLayer");
101}
