19 arm_compute::Status ClFullyConnectedWorkloadValidate(const TensorInfo& input,

29 
30  arm_compute::TensorInfo aclBiases;

43  optionalAclBiases,

100 
101  InitializeArmComputeClTensorData(*m_WeightsTensor, m_Data.m_Weight);
102
