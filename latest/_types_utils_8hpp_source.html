326 {
327  os << "[";
328  if (shape.GetDimensionality() != Dimensionality::NotSpecified)
329  {
330  for (uint32_t i = 0; i < shape.GetNumDimensions(); ++i)
331  {
332  if (i != 0)
333  {
334  os << ",";
335  }
336  if (shape.GetDimensionSpecificity(i))
337  {
338  os << shape[i];
339  }
340  else
341  {
342  os << "?";
343  }
344  }
345  }
346  else
347  {
348  os << "Dimensionality Not Specified";
349  }
350  os << "]";
